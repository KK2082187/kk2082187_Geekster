    double sub_2 = scn.nextDouble();
    double sub_3 = scn.nextDouble();
    double sub_4 = scn.nextDouble();
    double sub_5 = scn.nextDouble();
    double total = scn.nextDouble();
    double sum = sub_1+sub_2+sub_3+sub_4+sub_5;

    System.out.println(sum);
    System.out.println(percentage);
	}
}
